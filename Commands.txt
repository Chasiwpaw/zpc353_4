Create table Bookstore (name varchar(20) NOT NULL, email varchar(20), Primary key (name));

Create table Address (email varchar(20) NOT NULL, province varchar(10), city varchar(10), postalcode varchar(6), Primary key (email));

Create table Book (isbn varchar(10) NOT NULL, title varchar(20), edition int, author varchar(20), publisher varchar(20), price double, quantity_on_hand int, quantity_sold int, Primary key (isbn));

Create table Branch (branch_name varchar(20) NOT NULL, publisher varchar(20), Primary key (branch_name));

Create table Publisher (publisher_number int NOT NULL, company_name varchar(20), Primary key (publisher_number));

Create table `Order` (order_number varchar(10) NOT NULL, tracking_number varchar(10), confirmation_number varchar(10), date_placed datetime, Primary key (order_number));

Create table Sale (sale_number varchar(10) NOT NULL, date_sold datetime, Primary key (sale_number));

Create table Shipment (shipment_number varchar(10) NOT NULL, date_ship datetime, Primary key (shipment_number));

Create table Customer (name varchar(20), phone_number varchar(10) NOT NULL, Primary key (phone_number)); 

Create table Bookstore_employee (id varchar(10) NOT NULL, SSN varchar(9), name varchar(20), phone_number varchar(10), branch_representative varchar(20), branch_manager varchar(20));
